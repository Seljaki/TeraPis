x Statements = Statement Statements | Statement
x Statement = plot PlotDefinition | work WorkDefinition | if If | Function

x PlotDefinition = name { PlotBody }

x PlotBody = PlotBody' , PlotBody | PlotBody'
x PlotBody' = coordinates Coordinates | type PlotType

x Coordinates = : [ Points ]
x Points = Point , Points | Point
x Point = point ( REAL , REAL )

x PlotType =  : forest | : field


x WorkDefinition =  name { WorkBody }

x WorkBody = WorkBody' , WorkBody | WorkBody'
x WorkBody' = path Path | action Action | max-speed MaxSpeed | implement-width ImplementWidth | plot WorkPlot

x Path =  : [ Pointts ]
x Pointts = Pointt , Pointts | Pointt
x Pointt = pointt ( REAL , REAL , timestamp )

X Action =  : name
X MaxSpeed =  : REAL
X ImplementWidth = : REAL
X WorkPlot = : name


x If = plot name If'
x If' = is IfIsValid | contains IfContains
x IfIsValid =  valid { Statements }
x IfContains =  name { Statements }


Function = calculatepath ( name , name )
    | calculatearea ( name )
    | calculateefficiency ( name , name )
    | calculateareacovered ( name )
    | calculateaveragespeed ( name )
