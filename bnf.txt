x Statements = Statement Statements | Statement
Statement = PlotDefinition | WorkDefinition | If | Function

x PlotDefinition = plot name { PlotBody }

x PlotBody = PlotBody' , PlotBody | PlotBody'
x PlotBody' = Coordinates | PlotType

x Coordinates = coordinates : [ Points ]
x Points = Point , Points | Point
x Point = point ( REAL , REAL )

x PlotType = type : PlotType'
x PlotType' = forest | field


x WorkDefinition = work name { WorkBody }

x WorkBody = WorkBody' , WorkBody | WorkBody'
x WorkBody' = Path | Action | MaxSpeed | ImplementWidth | WorkPlot

x Path = path : [ Pointts ]
x Pointts = Pointt , Pointts | Pointt
x Pointt = pointt ( REAL , REAL , timestamp )

x Action = action : name
x MaxSpeed = max-speed : REAL
x ImplementWidth = implement-width : REAL
x WorkPlot = plot : name


x If = if plot name If'
x If' = IfIsValid | IfContains
x IfIsValid = is valid { Statements }
x IfContains = contains name { Statements }

// Functions = Function Functions | Function
x Function = CalculatePath | CalculateArea | CalculateEfficiency | CalculateAreaCovered | CalculateAverageSpeed

x CalculatePath = calculatepath ( name , name )
x CalculateArea = calculatearea ( name )
x CalculateEfficiency = calculateefficiency ( name , name )
x CalculateAreaCovered = calculateareacovered ( name )
x CalculateAverageSpeed = calculateaveragespeed ( name )
